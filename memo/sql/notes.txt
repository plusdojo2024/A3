CREATE TABLE NOTES (
  NOTE_ID	INTEGER  NOT NULL  AUTO_INCREMENT,
  FAMILY_ID	INTEGER NOT NULL AUTO_INCREMENT,
  TITLE		VARCHAR(20),
  NOTE		VARCHAR(200),
  NOTE_DATE	VARCHAR(32),
  NOTE_DATE_YEARMONTH VARCHAR(32),
  IMAGE_ONE	VARCHAR(64),
  IMAGE_TWO	VARCHAR(64),
  NOTE_UPDATE	VARCHAR(32),
  PRIMARY KEY (NOTE_ID),
  CONSTRAINT fk_notes_family_id
  FOREIGN KEY (family_id) REFERENCES users (family_id)
  ON DELETE SET NULL ON UPDATE CASCADE
);
