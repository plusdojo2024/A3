CREATE TABLE NOTES (
  NOTE_ID	INTEGER  NOT NULL  AUTO_INCREMENT,
  MAIL		VARCHAR(64),
  TITLE		VARCHAR(20),
  NOTE		VARCHAR(200),
  NOTE_DATE	VARCHAR(32),
  IMAGE_ONE	VARCHAR(64),
  IMAGE_TWO	VARCHAR(64),
  NOTE_UPDATE	VARCHAR(32),
  PRIMARY KEY (NOTE_ID)
  CONSTRAINT fk_notes_mail
  FOREIGN KEY (mail) REFERENCES users (mail)
  ON DELETE SET NULL ON UPDATE CASCADE
);
