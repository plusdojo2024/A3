
CREATE TABLE TODO_LIST(
    list_id INTEGER NOT NULL,
    mail VARCHAR,
    task VARCHAR(20) UNIQUE,
    category VARCHAR,
    give_point INTEGER(2),
    list_date VARCHAR(32),
    memo VARCHAR(200),
    todo_delete INTEGER,
    PRIMARY KEY (list_id),
    CONSTRAINT fk_todo_list_family_mail
    FOREIGN KEY (mail) REFERENCES (mail)
    ON DELETE SET NULL ON UPDATE CASCADE
)
