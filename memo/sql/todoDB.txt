
CREATE TABLE TODO (
  todo_id INTEGER NOT NULL AUTO_INCREMENT,
  list_id INTEGER,
  todo_date VARCHAR(32),
  uid INTEGER NOT NULL AUTO_INCREMENT,
  loop INTEGER DEFAULT 0,
  start_date VARCHAR,
  end_date VARCHAR,
  monday INTEGER(1) DEFAULT 0,
  tuesday INTEGER(1) DEFAULT 0,
  wednesday INTEGER(1) DEFAULT 0,
  thursday INTEGER(1) DEFAULT 0,
  friday INTEGER(1) DEFAULT 0,
  saturday INTEGER(1) DEFAULT 0,
  sunday INTEGER(1) DEFAULT 0,
  todo_complete INTEGER DEFAULT 0,
  family_update TIMESTAMP,
  PRIMARY KEY (todo_id),
  CONSTRAINT fk_todo_list_id
  FOREIGN KEY (list_id) REFERENCES todo_list (list_id)
  ON DELETE SET NULL ON UPDATE CASCADE,
  CONSTRAINT fk_todo_user_id
  FOREIGN KEY (uid) REFERENCES users (uid)
  ON DELETE SET NULL ON UPDATE CASCADE
  /*参照先がDELETEされたらNULL,UPDATEされたらそれを反映する*/
);
